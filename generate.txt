### Form Pattern with Zod Validation:
```typescript
"use client";

import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { createComicSchema } from "@/lib/validations/schemas";
import { createComic } from "@/lib/actions/comic";
import { Form, FormField, FormItem, FormLabel, FormControl, FormMessage } from "ui/form";
import { Input } from "ui/input";
import { Textarea } from "ui/textarea";
import { Button } from "ui/button";
import { useRouter } from "next/navigation";
import { toast } from "sonner";

export function ComicForm({ initialData }: { initialData?: any }) {
  const router = useRouter();
  const form = useForm({
    resolver: zodResolver(createComicSchema),
    defaultValues: initialData || {
      title: "",
      description: "",
      coverImage: "",
      status: "Ongoing",
    },
  });

  async function onSubmit(data: any) {
    const formData = new FormData();
    Object.entries(data).forEach(([key, value]) => {
      formData.append(key, String(value));
    });

    const result = await createComic(formData);
    
    if (result.error) {
      toast.error(result.error);
    } else {
      toast.success("Comic created successfully");
      router.push("/admin/comics");
      router.refresh();
    }
  }

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
        {/* Form fields */}
      </form>
    </Form>
  );
}
```

## üå± PHASE 4: DYNAMIC SEED HELPERS

### Enhanced seedHelpers.ts Structure:
```typescript
// src/lib/seedHelpers.ts
import { z } from "zod";
import { db } from "@/db";
import * as schema from "@/db/schema";
import { readFileSync } from "fs";
import { join } from "path";

// Generic JSON loader with validation
export async function loadJsonData<T>(
  filePath: string,
  validator: z.ZodSchema<T>
): Promise<T[]> {
  try {
    const fullPath = join(process.cwd(), filePath);
    const fileContent = readFileSync(fullPath, "utf-8");
    const jsonData = JSON.parse(fileContent);
    
    // Validate array of data
    const arraySchema = z.array(validator);
    return arraySchema.parse(jsonData);
  } catch (error) {
    console.error(`Error loading ${filePath}:`, error);
    throw error;
  }
}

// Dynamic seeder for any table
export async function seedTable<T extends keyof typeof schema>(
  tableName: T,
  data: any[],
  options: {
    truncate?: boolean;
    batchSize?: number;
  } = {}
) {
  const table = schema[tableName];
  const { truncate = false, batchSize = 100 } = options;

  if (truncate) {
    await db.delete(table);
  }

  // Insert in batches
  for (let i = 0; i < data.length; i += batchSize) {
    const batch = data.slice(i, i + batchSize);
    await db.insert(table).values(batch);
    console.log(`Inserted ${Math.min(i + batchSize, data.length)}/${data.length} records into ${String(tableName)}`);
  }
}

// Pattern-based file loader (e.g., chaptersdata*.json)
export async function loadJsonPattern(
  pattern: string,
  validator: z.ZodSchema
): Promise<any[]> {
  const glob = await import("glob");
  const files = glob.sync(pattern);
  
  let allData: any[] = [];
  for (const file of files) {
    const data = await loadJsonData(file, validator);
    allData = allData.concat(data);
  }
  
  return allData;
}
```

### Usage in seed.ts:
```typescript
import { loadJsonData, loadJsonPattern, seedTable } from "@/lib/seedHelpers";
import { userSchema, comicSchema, chapterSchema } from "@/lib/validations/seed";

async function main() {
  // Load from specific files
  const users = await loadJsonData("./users.json", userSchema);
  await seedTable("user", users, { truncate: true });

  // Load from pattern (all comicsdata*.json files)
  const comics = await loadJsonPattern("./comicsdata*.json", comicSchema);
  await seedTable("comic", comics);

  // Load chapters from pattern
  const chapters = await loadJsonPattern("./chaptersdata*.json", chapterSchema);
  await seedTable("chapter", chapters);
}
```

#### 3. Common Issues and Fixes:

**Issue: process.env usage outside app-config**
```typescript
// ‚ùå Bad
const apiKey = process.env.API_KEY;

// ‚úÖ Good
import { env } from "app-config";
const apiKey = env.API_KEY;
```

**Issue: Missing type definitions**
```typescript
// ‚ùå Bad
const data: any = await fetch(...);

// ‚úÖ Good
const data: ApiResponse<Comic[]> = await fetch(...);
```

**Issue: Unhandled promise rejections**
```typescript
// ‚ùå Bad
async function getData() {
  const data = await fetch(...);
  return data;
}

// ‚úÖ Good
async function getData(): Promise<ApiResponse<Data>> {
  try {
    const data = await fetch(...);
    return { success: true, data };
  } catch (error) {
    console.error("Error fetching data:", error);
    return { success: false, error: "Failed to fetch data" };
  }
}
```

**Issue: Missing nullish coalescing**
```typescript
// ‚ùå Bad
const value = data.field;

// ‚úÖ Good
const value = data.field ?? defaultValue;
```
