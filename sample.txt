```tsx
// src/components/comics/BookmarkActions.tsx
'use client';

import { useState, useTransition } from 'react';
import { addToBookmarksAction, removeFromBookmarksAction } from '@/lib/actions/bookmarks';
import { Button } from '@/components/ui/button';
import { BookmarkIcon, BookmarkFilledIcon } from '@/components/icons';
import { toast } from 'sonner';

interface BookmarkActionsProps {
  comicId: number;
  comicSlug: string;
  isBookmarked: boolean;
  initialStatus?: 'Reading' | 'PlanToRead' | 'Completed' | 'Dropped';
}

export function BookmarkActions({ comicId, comicSlug, isBookmarked, initialStatus }: BookmarkActionsProps) {
  const [bookmarked, setBookmarked] = useState(isBookmarked);
  const [isPending, startTransition] = useTransition();

  const handleAddBookmark = () => {
    startTransition(async () => {
      const result = await addToBookmarksAction({ comicId, status: 'Reading' });
      if (result.success) {
        setBookmarked(true);
        toast.success('Added to bookmarks!');
      } else {
        toast.error(result.error || 'Failed to add bookmark');
      }
    });
  };

  const handleRemoveBookmark = () => {
    startTransition(async () => {
      const result = await removeFromBookmarksAction({ comicId });
      if (result.success) {
        setBookmarked(false);
        toast.success('Removed from bookmarks!');
      } else {
        toast.error(result.error || 'Failed to remove bookmark');
      }
    });
  };

  return (
    <div className="flex items-center gap-2">
      {bookmarked ? (
        <Button
          variant="outline"
          onClick={handleRemoveBookmark}
          disabled={isPending}
        >
          <BookmarkFilledIcon className="mr-2 h-4 w-4" />
          Bookmarked
        </Button>
      ) : (
        <Button
          onClick={handleAddBookmark}
          disabled={isPending}
        >
          <BookmarkIcon className="mr-2 h-4 w-4" />
          Add to Bookmarks
        </Button>
      )}
    </div>
  );
}
```

**Convert all `any` types to:**
- Specific types from usage context
- Generic types where applicable
- Proper type inference
- Interface definitions

**Example:**
```typescript
// Before
function processData(data: any): any {
  return data.map((item: any) => item.value);
}

// After
function processData<T extends { value: unknown }>(data: T[]): unknown[] {
  return data.map((item) => item.value);
}
```
